[lo, ...ALIASES[lo]] : [lo]; } function buildFtsQuery(rawTerms: string[]): string { // FTS5 query: cada término con sus aliases en OR, todos los términos en AND return rawTerms.map(term => { const expanded = expandTerm(term); if (expanded.length === 1) return `"${expanded[0]}"*`; return '(' + expanded.map(a => `"${a}"*`).join(' OR ') + ')'; }).join(' AND '); } export default { async fetch(request: Request, env: Env): Promise { if (request.method === 'OPTIONS') { return new Response(null, { status: 204, headers: { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, OPTIONS', 'Access-Control-Allow-Headers': '*', } }); } const url = new URL(request.url); const path = url.pathname; if (path === '/search') { const q = (url.searchParams.get('q') || '').trim(); const page = Math.max(1, parseInt(url.searchParams.get('page') || '1')); const limit = Math.min(parseInt(url.searchParams.get('limit') || '500'), 1000); const offset = (page - 1) * limit; if (q.length < 2) { return new Response(JSON.stringify({ products: [], total: 0 }), { headers: corsHeaders }); } const rawTerms = q.toLowerCase().split(/\s+/).filter(t => t.length >= 2); const ftsQuery = buildFtsQuery(rawTerms); try { // Usar FTS5 para búsqueda rápida, luego JOIN con products para obtener todos los campos const countResult = await env.DB.prepare( `SELECT COUNT(*) as total FROM products_fts WHERE products_fts MATCH ?` ).bind(ftsQuery).first<{ total: number }>(); const results = await env.DB.prepare( `SELECT p.id, p.name, p.price, p.currency, p.image, p.url, p.store, p.brand, p.season, p.condition, p.sizes FROM products_fts fts JOIN products p ON fts.rowid = p.rowid WHERE products_fts MATCH ? ORDER BY p.price ASC LIMIT ? OFFSET ?` ).bind(ftsQuery, limit, offset).all(); const products = results.results.map((p: any) => ({ ...p, sizes: (() => { try { return JSON.parse(p.sizes || '["One size"]'); } catch { return ['One size']; } })(), })); return new Response(JSON.stringify({ products, total: countResult?.total || 0, page, limit, }), { headers: corsHeaders }); } catch (e: any) { // Fallback a LIKE si FTS falla const conditions: string[] = []; const params: string[] = []; for (const term of rawTerms) { const expanded = expandTerm(term); const orParts = expanded.map(() => `LOWER(name) LIKE ?`).join(' OR '); conditions.push(`(${orParts})`); expanded.forEach(alias => params.push(`%${alias}%`)); } const whereClause = conditions.join(' AND '); const results = await env.DB.prepare( `SELECT id, name, price, currency, image, url, store, brand, season, condition, sizes FROM products WHERE ${whereClause} ORDER BY price ASC LIMIT ?
